# Installation

## Install 

```
$ npm install cloudflare-dns-update
```

{% hint style="info" %}
 Add flag -g if you want use CLI feature
{% endhint %}

### Usage

```javascript
const CloudflareDns = require('cloudflare-dns-update');
let dns = new CloudflareDdns({
    token: 'YOUR_TOKEN_HERE'
});
```

```javascript
const CloudflareDns = require('cloudflare-dns-update');
let dns = new CloudflareDdns({
    key: 'YOUR_KEY_HERE',
    email: 'your_email@example.com'
});
```
